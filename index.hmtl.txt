    const texto = `Hoje eu só quis fazer um pequeno detalhe pra você...

Eu sei que já te pedi desculpas,
mas também queria te dizer isso de um jeito mais bonito e mais calmo.

Eu entendi o meu erro.
Entendi que eu devia ter prestado mais atenção,
e quero que você saiba que não foi algo que eu fiz de propósito.

Naquele momento, eu só estava muito feliz de te ver.
Às vezes eu fico meio bobinho quando se trata de você...
e, por estar tão contente, eu realmente não captei direito a mensagem como deveria.

Depois eu percebi melhor o que aconteceu,
e entendi por que isso te chateou.

Então eu só queria te dizer, com sinceridade,
que vou ter mais cuidado com você e com as coisas que vêm de você.
Porque você me importa de verdade.

E além de tudo isso...
você é alguém que eu quero muito bem. 💖

Isso aqui não é pra te pressionar,
nem pra te cobrar nada.

É só um gesto simples,
porque eu gosto de você
e queria te deixar algo bonito.`;

    let i = 0;
    let escrevendo = false;

    function abrirCarta(){
      const carta = document.getElementById("carta");
      if(escrevendo) return;

      carta.style.display = "block";

      if(carta.innerHTML.trim() !== ""){
        carta.scrollIntoView({ behavior: "smooth", block: "start" });
        return;
      }

      escrevendo = true;
      escreverTexto();
    }

    function escreverTexto(){
      const carta = document.getElementById("carta");

      if(i < texto.length){
        const char = texto.charAt(i);

        if(char === "\n"){
          carta.innerHTML += "<br>";
        } else {
          carta.innerHTML += char;
        }

        i++;
        setTimeout(escreverTexto, 18);
      } else {
        escrevendo = false;
        mostrarResposta();
        carta.scrollIntoView({ behavior: "smooth", block: "start" });
      }
    }

    function trocarFrase(){
      const frases = [
        "Seu gatinho bobinho fez isso com carinho de verdade. 💖",
        "Tem pessoas que deixam tudo mais bonito só por existirem. 🌸",
        "Isso aqui foi feito sem pressão... só com afeto. 💕",
        "Eu só queria deixar um detalhezinho fofo pra você. ✨",
        "Mesmo meio bobito, meu carinho por você é sincero. 🌷",
        "Se isso te fizer sorrir um pouquinho, já valeu muito. 💗"
      ];

      const sorteada = frases[Math.floor(Math.random() * frases.length)];
      document.getElementById("fraseExtra").innerText = sorteada;
    }

    function mostrarEu(){
      const box = document.getElementById("euBox");
      box.style.display = "block";
      box.scrollIntoView({ behavior: "smooth", block: "center" });
    }

    function mostrarResposta(){
      const resposta = document.getElementById("resposta");
      resposta.style.display = "block";
      resposta.innerHTML = "Eu só queria te deixar algo leve, fofo e sincero... do jeitinho que o seu gatinho bobito conseguiu fazer. 💖";
    }
